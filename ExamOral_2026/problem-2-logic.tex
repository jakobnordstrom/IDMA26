

\begin{frame}
  \frametitle{Propositional Logic}

  Decide for each of the propositional logic formulas below whether it 
  is a tautology, contradiction, or neither

\blankline

  \begin{enumerate}
  \item 
    $
    (p \limpl q) \limpl (q \limpl p)
    $

\blankline

  \item
    $
    (p \land q)
    \lor
    (p \limpl \lnot q)
    $

  \end{enumerate}

\end{frame}
