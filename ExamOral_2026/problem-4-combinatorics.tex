\item 
  $4$ \textbf{suits:}
  $\heartsuit$,

  $\diamondsuit$

\item 
  $13$ \textbf{ranks:} $2$--$10$, jack, queen, king, and ace
\end{itemize}

A \textbf{poker hand}  contains $5$ cards.
%   Can you match the three poker hands below
%   with the correct probabilities of drawing them?
Match the three poker hands below
with the correct probabilities of drawing them!

\blankline

\blankline

\textbf{(1) Four of a kind:}
$4$ of the cards have the same rank

\blankline


\textbf{(2) Full house:}
$3$ cards of one rank,
$2$ cards of another rank

\blankline


\textbf{(3) Flush}
(including \textbf{straight flush})\textbf{:}
All cards are of the same suit

\blankline
