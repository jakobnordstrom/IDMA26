\begin{frame}
  \frametitle{Introduction to Discrete Mathematics\ldots}

  \textbf{Discrete}
  \begin{itemize}
  \item
    numbers, logic, graphs, combinations
  \item
    \textbf{\emph{not}}
    continuous functions, derivatives, integrals,
    et cetera
  \end{itemize}

  \bigskip
  
  \textbf{Mathematics}
  \begin{itemize}
  \item
    Start from obvious facts
    --- \alertred{axioms}
    
  \item
    \alertblue{Derive true facts} from these axioms
    
  \item
%       We won't give fully axiomatic treatment
%       (far from it)
    We'll be far from fully axiomatic treatment,
    but will stress \alertblue{rigorous reasoning}
   
  \item
    In mathematics, we establish
    \alertred{absolute truths}

  \item
    Very different from other fields of science
    using hypotheses and experiments
    
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{\ldots{} and Algorithms}

  
  \begin{itemize}
  \item
    Our computational problems have
    \alertblue{concrete, correct answers}
    (so no ML predictions)
    
  \item
    \alertred{Algorithms}
    are
    \alertblue{precise descriptions
      of automated methods}
    for solving such problems
    (focusing on discrete objects)
    
  \item
    Can be coded up in different programming languages
    
  \item
    But we won't have programming assignments in this course
    ---
    focus on descriptions in pseudocode
    rather than exact syntax
    
    
  \end{itemize}
\end{frame}
